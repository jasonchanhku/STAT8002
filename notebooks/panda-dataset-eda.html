Masks¶

  • Masks labels are stored in the Red channel of RGB. Hence, need to extract the matrix in the R channel
  • Map raw label values to RGB values

Radboud: Prostate glands are individually labelled. Valid values are:

  • 0: background (non tissue) or unknown
  • 1: stroma (connective tissue, non-epithelium tissue)
  • 2: healthy (benign) epithelium
  • 3: cancerous epithelium (Gleason 3)
  • 4: cancerous epithelium (Gleason 4)
  • 5: cancerous epithelium (Gleason 5)

Karolinska: Regions are labelled. Valid values are:

  • [0]: background (non tissue) or unknown
  • [1]: benign tissue (stroma and epithelium combined)
  • [2]: cancerous tissue (stroma and epithelium combined)

def preview_random_mask(resolution_lim=None):
    
    masks_list = glob(mask_dir + '/*.tiff')
    random_mask = masks_list[np.random.choice(len(masks_list))]
    sample_mask = openslide.OpenSlide(random_mask)
    
    mask_name = random_mask.split('/')[-1].split('_')[0]
    data_provider = label_df.loc[mask_name]['data_provider']
    isup = label_df.loc[mask_name]['isup_grade']
    
    mask_res = sample_mask.level_dimensions[-1]
    mask = sample_mask.read_region((0,0), level=2, size=mask_res)
    
    # extract the Red channel
    mask = mask.split()[0]
    
    # 256 * 3 = 768
    preview_palette = np.zeros(shape=768, dtype=int)
    
    if data_provider == 'radboud':
        # Mapping: {0: background, 1: stroma, 2: benign epithelium, 3: Gleason 3, 4: Gleason 4, 5: Gleason 5}
        # 0,0,0 -> black
        # 127,127,127 -> grey
        # 0,255,0 -> green
        # 255,255,178 -> yellow
        # 255,127,0 -> orange
        # 255,0,0 -> red
        
        preview_palette[0:18] = (np.array([  0,   0,   0, 127, 127, 127,   0, 255,   0, 255, 255, 178, 255,
       127,   0, 255,   0,   0]))
    elif data_provider == 'karolinska':
        # Mapping: {0: background, 1: benign, 2: cancer}
        # 0,0,0 -> black
        # 127,127,127 -> grey
        # 255,0,0 -> red
        preview_palette[0:9] = (np.array([  0,   0,   0, 127, 127, 127, 255,   0,   0]))
        
    mask.putpalette(data=preview_palette.tolist())
    mask = mask.convert(mode='RGB')
    
    if resolution_lim:
        mask.thumbnail(size=resolution_lim, resample=0)
        
    print(f"Mask name: {random_mask.split('/')[-1]}")
    print(f"ISUP Grade: {isup}")
    print(f"Data Provider: {data_provider}")
    print(f"Original resolution of Mask: {mask_res}")
    
    return display(mask)

def get_mask_labels_count(row):
    biopsy = skimage.io.MultiImage(mask_dir + '/' + row['image'] + '_mask.tiff')
    temp = biopsy[-1][:, :, 0]
    # value counts of the mask labels
    mask_label_counts = pd.Series(temp.reshape(-1)).value_counts()
    row.update(mask_label_counts)
    
    return row
